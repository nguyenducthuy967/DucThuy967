function countUppercaseChars(str) {
    let count = 0;
    for (let i = 0; i < str.length; i++) {
        if (str[i] >= 'A' && str[i] <= 'Z') {
            count++;
        }
    }

    if (count > 0) {
        return count;
    } else {
        return "The Strings is not contain upper char";
    }
}

// Ví dụ sử dụng:
console.log(countUppercaseChars("Hello World")); // Kết quả: 2
console.log(countUppercaseChars("hello world")); // Kết quả: The Strings is not contain upper char
console.log(countUppercaseChars("JAVASCRIPT")); // Kết quả: 10

function isPrime(num) {
    if (num <= 1) {
        return false;
    }
    for (let i = 2; i <= Math.sqrt(num); i++) {
        if (num % i === 0) {
            return false;
        }
    }
    return true;
}

function findPrimes(arr) {
    const primeNumbers = [];
    for (let i = 0; i < arr.length; i++) {
        if (isPrime(arr[i])) {
            primeNumbers.push(arr[i]);
        }
    }
    return primeNumbers;
}
function findAndSortPrimes(arr) {
    const primeNumbers = findPrimes(arr);
    primeNumbers.sort((a, b) => a - b);
    return primeNumbers;
}

const numbers = [10, 2, 7, 13, 5, 11, 4, 9, 17, 1];
const result = findAndSortPrimes(numbers);
console.log(result);

class DienThoai {
    constructor(maDienThoai, ten, hangSanXuat, gia) {
        this.maDienThoai = maDienThoai;
        this.ten = ten;
        this.hangSanXuat = hangSanXuat;
        this.gia = gia;
    }

    getMaDienThoai() { return this.maDienThoai; }
    getTen() { return this.ten; }
    getHangSanXuat() { return this.hangSanXuat; }
    getGia() { return this.gia; }

    setMaDienThoai(maDienThoai) { this.maDienThoai = maDienThoai; }
    setTen(ten) { this.ten = ten; }
    setHangSanXuat(hangSanXuat) { this.hangSanXuat = hangSanXuat; }
    setGia(gia) { this.gia = gia; }

    toString() {
        return `Mã: ${this.maDienThoai}, Tên: ${this.ten}, Hãng: ${this.hangSanXuat}, Giá: ${this.gia}`;
    }
}
